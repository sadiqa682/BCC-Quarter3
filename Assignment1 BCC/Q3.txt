    constructor() public{
        
    }
    
    string a = "Truck,Cars and Motorcycles are all vehciles";
    string b = " My car is servised";
    string c = "My Motor cycle is serviced";

    function A()  virtual public  {
                 emit logString(a);

    }
    function C()  virtual public  {
                 emit logString(c);
        
    }
}
